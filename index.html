                        function getDimensions() {
                            w1 = window.innerWidth;
                            h1 = window.innerHeight;
            
                            $(".image").css("width", (w1 - 100) + "px");
            
                            if (w1 > 1300) {
                                $(".image").css("width", 1200 + "px");
                                $(".image").css("margin-left", ((w1 - 1300) / 2) + "px");
                                $(".image").css("margin-right", ((w1 - 1300) / 2) + "px");
                            } else {
                                $(".image").css("width", (w1 - 100) + "px");
                                $(".image").css("margin-left", 0 + "px");
                                $(".image").css("margin-right", 0 + "px");
                            }
                        }
            
                        $(".image").css("height", "100px");
            
                        $(".image").hover(function () {
                            $(this).find(".description").fadeOut(1000);
                        }, function () {
                            $(this).find(".description").fadeIn();
                        });
            
                        $(".image").hover(function () {
                            if (w1 > 1300) {
                                $(this).css("height", (1200 * 0.56) + "px");

                                $(this).css("height", ((w1 - 100) * 0.56) + "px");
                            }
                        }, function () {
                            $(this).css("height", "100px");
                        });
            
                        getDimensions();
